def payload_storage(*args)
  unless(@_payload_dirs)
    @_payload_dirs = []
  end
  unless(args.empty?)
    @_payload_dirs += args.find_all do |dir_path|
      File.directory?(dir_path)
    end
    @_payload_dirs.uniq!
  end
  @_payload_dirs
end
